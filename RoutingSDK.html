require([ "esri/Map", "esri/views/MapView", "esri/tasks/RouteTask", "esri/tasks/support/RouteParameters", "esri/tasks/support/FeatureSet", "esri/layers/FeatureLayer", "esri/Graphic" ], function(Map, MapView, RouteTask, RouteParameters, FeatureSet, FeatureLayer, Graphic) { const map = new Map({ basemap: "streets-navigation-vector" }); const view = new MapView({ container: "viewDiv", map: map, center: [44.35350070376542, -78.74056278446773], // Fleming College coordinates zoom: 12 }); view.when(() => { // Load the feature layer for stops from ArcGIS Online const stopsLayer = new FeatureLayer({ url: "https://services.arcgis.com/V6ZHFr6zdgNZuVG0/arcgis/rest/services/Directions_Stops/FeatureServer/0" }); // Add the stops layer to the map map.add(stopsLayer); // Create a RouteTask using the ArcGIS Online routing service const routeTask = new RouteTask({ url: "https://services1.arcgis.com/pMeXRvgWClLJZr3s/arcgis/rest/services/Lindsay_Transit_Routes/FeatureServer" }); // Set up the route parameters stopsLayer.when(() => { const stops = new FeatureSet({ features: stopsLayer.source.items.map(item => new Graphic({ geometry: item.geometry })) }); const params = new RouteParameters({ stops: stops, returnDirections: true }); // Solve the route using the route task routeTask.solve(params).then(result => { // Handle the routing result console.log(result); }).catch(error => { // Handle routing error console.error(error); }); }); }); });
